CREATE TABLE tags_registed(
	id int NOT NULL AUTO-INCREMENT,
	tag varchar(255),
	PRIMARY KEY (id)	
);

CREATE TABLE user(
	tag_id int NOT NULL,
	name varchar(255),
	FOREIGN KEY (tag_id) REFERENCES tags_registed(id)
);

CREATE TABLE in_out(
    -> tag_id int NOT NULL,
    -> time_in datetime,
    -> time_out datetime,
    -> FOREIGN KEY (tag_id) REFERENCES tags_registed(id));
